-- ✅ Fixed Yukio Hub v2 (Stable)
game:IsLoaded()

local WindUI = loadstring(game:HttpGet('https://github.com/Footagesus/WindUI/releases/latest/download/main.lua'))()

local Players = game:GetService('Players')
local RunService = game:GetService('RunService')
local Workspace = game:GetService('Workspace')

local LocalPlayer = Players.LocalPlayer
local Camera = Workspace.CurrentCamera

-- Settings
local Settings = {
    FOV = 70,
    WalkSpeed = 16,
    JumpHeight = 7.2,
    ShowShootBtn = false,
    ShowKillAllBtn = false,
}

local Window = WindUI:CreateWindow({
    Folder = 'YukioHub',
    ToggleKey = Enum.KeyCode.LeftShift,
    Title = 'Yukio Hub',
    Transparent = true,
    Theme = 'Red',
    Author = 'Fixed Version',
    Size = UDim2.fromOffset(560, 440),
})

-- Create Floating Buttons
local ScreenGui = Instance.new("ScreenGui")
ScreenGui.ResetOnSpawn = false
ScreenGui.Parent = LocalPlayer:WaitForChild("PlayerGui")

local function MakeButton(name, text, position)
    local btn = Instance.new("TextButton")
    btn.Name = name
    btn.Size = UDim2.fromOffset(100, 55)
    btn.Position = position
    btn.BackgroundColor3 = Color3.fromRGB(30, 30, 30)
    btn.BackgroundTransparency = 0.3
    btn.Text = text
    btn.TextColor3 = Color3.new(1,1,1)
    btn.Font = Enum.Font.GothamBold
    btn.TextSize = 15
    btn.Visible = false
    btn.Parent = ScreenGui

    Instance.new("UICorner", btn).CornerRadius = UDim.new(0, 12)
    local stroke = Instance.new("UIStroke", btn)
    stroke.Thickness = 2
    stroke.Color = Color3.fromRGB(139, 0, 0)
    
    return btn
end

local ShootButton = MakeButton("ShootBtn", "🔫 Shoot", UDim2.new(0.35, 0, 0.45, 0))
local KillButton = MakeButton("KillBtn", "💀 Kill All", UDim2.new(0.65, 0, 0.45, 0))

-- Button Functions
ShootButton.MouseButton1Click:Connect(function()
    local gun = LocalPlayer.Backpack:FindFirstChild("Gun") or (LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("Gun"))
    if not gun then WindUI:Notify({Title="Error", Content="No Gun found!", Duration=3}) return end
    
    gun.Parent = LocalPlayer.Character
    task.wait()
    local shoot = gun:FindFirstChild("Shoot")
    if shoot then
        for _, v in ipairs(Players:GetPlayers()) do
            if v \~= LocalPlayer and v.Character and v.Character:FindFirstChild("HumanoidRootPart") then
                local root = v.Character.HumanoidRootPart
                shoot:FireServer(LocalPlayer.Character.HumanoidRootPart.CFrame, root.CFrame + root.Velocity * 0.08)
            end
        end
    end
    gun.Parent = LocalPlayer.Backpack
end)

KillButton.MouseButton1Click:Connect(function()
    local knife = LocalPlayer.Character:FindFirstChild("Knife") or LocalPlayer.Backpack:FindFirstChild("Knife")
    if not knife then WindUI:Notify({Title="Error", Content="Need Knife!", Duration=3}) return end

    for _, v in ipairs(Players:GetPlayers()) do
        if v \~= LocalPlayer and v.Character and v.Character:FindFirstChild("HumanoidRootPart") then
            local hrp = v.Character.HumanoidRootPart
            hrp.Anchored = true
            hrp.CFrame = LocalPlayer.Character.HumanoidRootPart.CFrame * CFrame.new(0, 0, -3)
        end
    end

    local stab = knife:FindFirstChild("Stab")
    if stab then stab:FireServer("Slash") end

    task.wait(0.3)
    for _, v in ipairs(Players:GetPlayers()) do
        if v.Character and v.Character:FindFirstChild("HumanoidRootPart") then
            v.Character.HumanoidRootPart.Anchored = false
        end
    end
end)

-- Tabs
local Main = Window:Tab({Title = 'Main', Icon = 'circle-user-round'})
local Sheriff = Window:Tab({Title = 'Sheriff', Icon = 'shield'})
local Murderer = Window:Tab({Title = 'Murderer', Icon = 'sword'})

Main:Slider({
    Title = "FOV",
    Min = 60,
    Max = 120,
    Default = 70,
    Callback = function(v)
        Settings.FOV = v
        Camera.FieldOfView = v
    end
})

Main:Slider({
    Title = "WalkSpeed",
    Min = 16,
    Max = 120,
    Default = 16,
    Callback = function(v)
        Settings.WalkSpeed = v
        local hum = LocalPlayer.Character and LocalPlayer.Character:FindFirstChildOfClass("Humanoid")
        if hum then hum.WalkSpeed = v end
    end
})

Main:Slider({
    Title = "Jump Height",
    Min = 7,
    Max = 100,
    Default = 7.2,
    Callback = function(v)
        Settings.JumpHeight = v
        local hum = LocalPlayer.Character and LocalPlayer.Character:FindFirstChildOfClass("Humanoid")
        if hum then hum.JumpHeight = v end
    end
})

Sheriff:Toggle({
    Title = "Show Shoot Button",
    Default = false,
    Callback = function(v)
        Settings.ShowShootBtn = v
        ShootButton.Visible = v
    end
})

Murderer:Toggle({
    Title = "Show Kill All Button",
    Default = false,
    Callback = function(v)
        Settings.ShowKillAllBtn = v
        KillButton.Visible = v
    end
})

-- Respawn Handler
LocalPlayer.CharacterAdded:Connect(function()
    task.wait(1)
    local hum = LocalPlayer.Character:WaitForChild("Humanoid")
    hum.WalkSpeed = Settings.WalkSpeed
    hum.JumpHeight = Settings.JumpHeight
end)

WindUI:Notify({
    Title = "Yukio Hub",
    Content = "Successfully Loaded! Press Left Shift",
    Duration = 6
})

print("✅ Yukio Hub Fixed & Loaded!")
